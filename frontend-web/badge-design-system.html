BadgeButton — Componente Bottone

// components/ui/BadgeButton.tsx
import React from 'react'
import { ButtonBase, CircularProgress } from '@mui/material'

type Variant = 'primary' | 'secondary' | 'ghost' | 'danger'
type Size    = 'sm' | 'md' | 'lg'

interface BadgeButtonProps {
  variant?:  Variant
  size?:     Size
  loading?: boolean
  icon?:    React.ReactNode
  children: React.ReactNode
  onClick?: () => void
  disabled?: boolean
}

const variantStyles: Record<Variant, string> = {
  primary:   'bg-badge-navy-500 text-white hover:bg-badge-navy-700 shadow-sm hover:shadow-md',
  secondary: 'bg-transparent text-badge-navy-500 border border-badge-navy-200 hover:bg-badge-navy-50',
  ghost:     'bg-transparent text-badge-stone border border-badge-bone hover:bg-badge-parchment',
  danger:    'bg-red-700 text-white hover:bg-red-800',
}

const sizeStyles: Record<Size, string> = {
  sm: 'h-[34px] px-4 text-xs',
  md: 'h-[44px] px-5 text-sm',   // WCAG 2.5.5 touch target
  lg: 'h-[52px] px-8 text-base',
}

export function BadgeButton({
  variant = 'primary',
  size = 'md',
  loading = false,
  icon,
  children,
  onClick,
  disabled,
}: BadgeButtonProps) {
  return (
    <ButtonBase
      onClick={onClick}
      disabled={disabled || loading}
      className={`
        inline-flex items-center gap-2 rounded-lg font-['DM_Sans'] font-medium
        tracking-wide transition-all duration-200 ease-out
        active:translate-y-px select-none
        ${variantStyles[variant]}
        ${sizeStyles[size]}
        ${(disabled || loading) ? 'opacity-50 cursor-not-allowed pointer-events-none' : ''}
      `}
    >
      {loading ? (
        <CircularProgress size={14} color="inherit" />
      ) : icon}
      {!loading && children}
    </ButtonBase>
  )
}

KpiCard — Componente Card KPI

// components/dashboard/KpiCard.tsx
import { cn } from '@/lib/utils'

type KpiVariant = 'total' | 'present' | 'absent' | 'late'

interface KpiCardProps {
  label:    string
  value:    number
  delta?:   string
  trend?:   'up' | 'down' | 'neutral'
  variant:  KpiVariant
}

const accentMap: Record<KpiVariant, string> = {
  total:   'border-l-badge-navy-500',
  present: 'border-l-emerald-700',
  absent:  'border-l-red-700',
  late:    'border-l-amber-600',
}

const valueColorMap: Record<KpiVariant, string> = {
  total:   'text-badge-ink',
  present: 'text-emerald-700',
  absent:  'text-red-700',
  late:    'text-amber-600',
}

export function KpiCard({ label, value, delta, trend, variant }: KpiCardProps) {
  return (
    <div className={cn(
      'bg-white rounded-xl border border-badge-bone p-6',
      'border-l-[3px] transition-shadow hover:shadow-md',
      accentMap[variant]
    )}>
      <p className="text-[11px] font-semibold tracking-[0.12em] uppercase text-badge-stone mb-3">
        {label}
      </p>
      <p className={cn(
        'font-["Cormorant"] text-[64px] font-light leading-none tracking-[-0.03em] mb-2',
        valueColorMap[variant]
      )}>
        {value}
      </p>
      {delta && (
        <p className={cn(
          'text-xs text-badge-stone',
          trend === 'up'   && 'text-emerald-700',
          trend === 'down' && 'text-red-700'
        )}>
          {delta}
        </p>
      )}
    </div>
  )
}

StatusBadge — Badge Presenza con Accessibilità

Presente Assente In ritardo 
// components/ui/StatusBadge.tsx
type Status = 'present' | 'absent' | 'late' | 'unknown'

const statusConfig: Record<Status, { label: string; classes: string; dot: string }> = {
  present: { label: 'Presente',   classes: 'bg-emerald-50 text-emerald-800',  dot: 'bg-emerald-700' },
  absent:  { label: 'Assente',    classes: 'bg-red-50 text-red-800',        dot: 'bg-red-700' },
  late:    { label: 'In ritardo', classes: 'bg-amber-50 text-amber-800',     dot: 'bg-amber-600' },
  unknown: { label: 'Non rilevato', classes: 'bg-stone-100 text-stone-600', dot: 'bg-stone-400' },
}

export function StatusBadge({ status }: { status: Status }) {
  const { label, classes, dot } = statusConfig[status]
  return (
    <span
      role="status"
      aria-label={`Stato presenza: ${label}`}
      className={`inline-flex items-center gap-1.5 px-2.5 py-0.5 rounded-full text-xs font-medium ${classes}`}
    >
      <span className={`w-1.5 h-1.5 rounded-full ${dot}`} aria-hidden="true" />
      {label}
    </span>
  )
}
